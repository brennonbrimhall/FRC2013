102    public void robotInit() {
103        driverStick = new XboxController(1);
104        operatorStick = new LogitechDualActionController(2);
105
106        compressor = new Compressor(1, 1);
107        compressor.start();
108
109        fl = new Talon(1);
110        bl = new Talon(2);
111        fr = new Talon(3);
112        br = new Talon(4);
113
114        gyro = new Gyro(1);
115
116        driveEncoder = new Encoder(6, 7);
117
118        drivetrain = new Drivetrain(fl, bl, fr, br, gyro, driveEncoder);
119
120        shooterTalon = new Talon(5);
121        beltTalon = new Talon(6);
122        collectorTalon = new Talon(7);
123        traySolenoid = new DoubleSolenoid(1, 2);
124        latchSolenoid = new DoubleSolenoid(3, 4);
125        indexerSolenoid = new DoubleSolenoid(5, 6);
126        shooterEncoder = new Encoder(4, 5);
127        shooterEncoder.start();
128
129        tray = new Tray(shooterTalon, beltTalon, collectorTalon, traySolenoid, latchSolenoid, indexerSolenoid, shooterEncoder);
130        statusPrinter = new StatusPrinter(tray);

132        cycleCounter = 0;
133
134        lifterSolenoid = new DoubleSolenoid(7, 8);
135        leftLimit = new DigitalInput(2);
136        rightLimit = new DigitalInput(3);
137
138        lifters = new Lifters(lifterSolenoid, leftLimit, rightLimit);
139
140        leftLight = new Relay(2);
141        rightLight = new Relay(3);

145    public void autonomousInit() {
146        //Lowering tray; setting up for all auto modes
147        init();
148
149        //Setting cycle counter; will use in various auto modes
150        cycleCounter = 0;
151
152        //Deciding autonomous mode
153        decideAutonomous();
154    }

156    /**
157     * This function is called periodically during autonomous
158     */
159    public void autonomousPeriodic() {
160        //Incrementing cycles
161        cycleCounter++;
162
163        //Deciding autonomous function call based on autoMode using a switch or
164        //break setup to improve speed.
165        switch (autoMode) {
166            case kTwoDiskAuto:
167                twoDiskAutonomousPeriodic();
168                break;
169            case kThreeDiskAuto:
170                threeDiskAutonomousPeriodic();
171                break;
172            case kFourDiskAuto:
173                fourDiskAutonomousPeriodic();
174                break;
175            case kNoAuto:
176                break;
177        }
178
179        //Updating tray
180        tray.update();
181    }
182
183    private void decideAutonomous() {
184        //Setting two disk auto if DIO 2 is on
185        if (DriverStation.getInstance().getDigitalIn(2)) {
186            autoMode = kTwoDiskAuto;
187            //Setting three disk auto if DIO 3 is on
188        } else if (DriverStation.getInstance().getDigitalIn(3)) {
189            autoMode = kThreeDiskAuto;
190            //Setting four disk auto if DIO 4 is on
191        } else if (DriverStation.getInstance().getDigitalIn(4)) {
192            autoMode = kFourDiskAuto;
193            //Defaulting to no autonomous if no DIO is on
194        } else {
195            autoMode = kNoAuto;
196        }
197    }
198
199    private void twoDiskAutonomousPeriodic() {
200        if (cycleCounter < 100) {
201            //Waiting for tray to come down, etc.
202        } else if (cycleCounter < 400) {
203            //Backing up to pyramid
204            if (!lifters.isOnPyramid()) {
205                drivetrain.safeArcadeDrive(-.15, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
206            } else {
207                drivetrain.safeArcadeDrive(0, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
208            }
209        } else if (cycleCounter < 600) {
210            //Shooting first disk
211            drivetrain.safeArcadeDrive(0, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
212            tray.shoot();
213        } else {
214            //Shooting second disk
215            drivetrain.safeArcadeDrive(0, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
216            tray.notShoot();
217        }
218        statusPrinter.printStatuses();
219    }
220
221    private void threeDiskAutonomousPeriodic() {
222        if (threeDiskAutoState == kThreeDiskAutoReverse) {
223            drivetrain.arcadeDrive(.5, 0);
224            if (drivetrain.getRightDistance() >= kThreeDiskAutoReverseDistance) {
225                threeDiskAutoState = kThreeDiskAutoTurn;
226            }
227        } else if (threeDiskAutoState == kThreeDiskAutoTurn) {
228            drivetrain.arcadeDrive(0, -.4);
229            if (drivetrain.getRightDistance() <= kThreeDiskAutoTurnDistance) {
230                threeDiskAutoState = kThreeDiskAutoFire;
231            }
232        } else if (threeDiskAutoState == kThreeDiskAutoFire) {
233            drivetrain.arcadeDrive(0, 0);
234            tray.shoot();
235            if (cycleCounter >= 650) {
236                threeDiskAutoState = kThreeDiskAutoWait;
237            }
238        } else if (threeDiskAutoState == kThreeDiskAutoWait) {
239            drivetrain.arcadeDrive(0, 0);
240            tray.notShoot();
241        }
242    }
243
244    private void fourDiskAutonomousPeriodic() {
245        //Lower tray
246        if (cycleCounter < 25) {
247            //Waiting for tray to lower
248        //Moving backwards
249        } else if (cycleCounter < 75) {
250            if (!lifters.isOnPyramid()) {
251                drivetrain.safeArcadeDrive(-.15, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
252            } else {
253                drivetrain.safeArcadeDrive(0, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
254            }
255        //Shooting first preloaded disk
256        } else if (cycleCounter < 150) {
257            drivetrain.safeArcadeDrive(0, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
258            tray.shoot();
259        //Shooting second preloaded disk
260        }else if (cycleCounter < 200) {
261            drivetrain.safeArcadeDrive(0, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
262            tray.notShoot();
263            //Resetting encoders and gyro for next legs of auto
264            drivetrain.resetGyro();
265            drivetrain.resetDistance();
266        //Driving forwards
267        }else if (cycleCounter < 250) {
268            if(drivetrain.getLeftDistance()<100 && drivetrain.getRightDistance()<100){
269                drivetrain.safeArcadeDrive(.5, 0, false, false);
270            }
271        //Turning 90 degrees counterclockwise
272        }else if (cycleCounter < 300) {
273            if(!(drivetrain.gyro.getAngle() > 95) && !(drivetrain.gyro.getAngle() < 85)) {
274                //do nothing, in correct orientation
275            }else{
276                drivetrain.safeArcadeDrive(.5, 1, false, false);
277            }
278        //Backing up and collecting
279        }else if (cycleCounter < 500) {
280            tray.collectorOn();
281            tray.beltOn();
282            if(drivetrain.getLeftDistance()<50 && drivetrain.getRightDistance()<50){
283                drivetrain.arcadeDrive(.5, 0);
284            }
285        //Turning back straight
286        }else if (cycleCounter < 550) {
287            if(!(drivetrain.gyro.getAngle() > 5) && !(drivetrain.gyro.getAngle() < -5)) {
288                //do nothing, in correct orientation
289            }else{
290                drivetrain.safeArcadeDrive(.5, 1, false, false);
291            }
292        //Driving backwards
293        }else if (cycleCounter < 600) {
294            if(!lifters.isOnPyramid()){
295                drivetrain.safeArcadeDrive(-.5, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
296            }
297        //Shooting first picked up disk
298        } else if (cycleCounter < 650) {
299            drivetrain.safeArcadeDrive(0, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
300            tray.shoot();
301        //Shooting second picked up disk
302        }else if (cycleCounter < 725) {
303            drivetrain.safeArcadeDrive(0, 0, lifters.leftOnPyramid(), lifters.rightOnPyramid());
304            tray.notShoot();
305        }
306        
307    }

319    /**
320     * This function is called periodically during operator control
321     */
322    public void teleopPeriodic() {
323        // Turn the lights on and off
324        lights();
325        
326        tray.setShooterVoltage();
327        driveOI();
328        trayRaiseOI();
329        collectorOI();
330        beltOI();
331        liftersOI();
332        
333        allOffOI();
334        shooterOnOI();
335        limitSwitchesOI();
336        shootOI();
337        
338        //Updating tray
339        tray.update();
340        
341        //Updating statuses
342        statusPrinter.printStatuses();
343
344    }

346    private void trayRaiseOI(){
347        // Raise and slower the tray using A and B buttons
348        if (operatorStick.getButton(6)) {
349            tray.raise();
350        }
351        if (operatorStick.getButton(8)) {
352            tray.lower();
353        }
354    }
355    
356    private void collectorOI(){
357        //Collector on, off, and reverse
358        if (operatorStick.getButton(4)) {
359            tray.collectorOn();
360        }
361        if (operatorStick.getButton(3) || operatorStick.getButton(1)) {
362            tray.collectorOff();
363        }
364        if (operatorStick.getButton(2)) {
365            tray.collectorReverse();
366        }
367    }
368    
369    private void beltOI(){
370        //Belt on, off, and reverse
371        if (operatorStick.dPadUp()) {
372            tray.beltOn();
373        }
374        if (operatorStick.dPadLeft() || operatorStick.dPadRight()) {
375            tray.beltOff();
376        }
377        if (operatorStick.dPadDown()) {
378            tray.beltReverse();
379        }
380    }
381    
382    private void liftersOI(){
383        //Raise and lower the lifters with left bumper and limit switches
384        //with right bumper override.
385        if (driverStick.getLeftBumper() && (driverStick.getRightBumper() || (lifters.isOnPyramid()))) {
386            lifters.lower();
387        }
388        if (driverStick.getB()) {
389            lifters.raise();
390        }
391    }
392    
393    private void driveOI(){
394        //Drive the robot
395        drivetrain.safeCheesyDrive(driverStick.getLeftY(), 
396                driverStick.getRighttX(), driverStick.getAnalogTriggers(),
397                lifters.leftOnPyramid(), lifters.rightOnPyramid());
398    }
399    
400    private void shootOI(){
401        //Shoot!
402        if (operatorStick.getButton(5) && (operatorStick.getButton(7) || (lifters.isOnPyramid()))) {
403            tray.shoot();
404        } else {
405            tray.notShoot();
406        }
407    }
408    
409    private void limitSwitchesOI(){
410        //Disabling limit switches
411        if (driverStick.getX()) {
412            lifters.disableSwitch();
413        }
414        
415        //Enabling limit switches 
416        if (driverStick.getY()) {
417            lifters.enableSwitch();
418        }
419    }
420    
421    private void allOffOI(){
422        //All off button - turns all tray functions off
423        if (operatorStick.getButton(9)) {
424            tray.allOff();
425        }
426    }
427    
428    private void shooterOnOI(){
429        //Turns shooter on
430        if (operatorStick.getButton(10)) {
431            tray.shooterOn();
432        }
433    }

470    private void lights(){
471        if (lifters.leftOnPyramid()) {
472            leftLight.set(Relay.Value.kForward);
473        } else {
474            leftLight.set(Relay.Value.kOff);
475        }
476        if (lifters.rightOnPyramid()) {
477            rightLight.set(Relay.Value.kForward);
478        } else {
479            rightLight.set(Relay.Value.kOff);
480        }
481    }
