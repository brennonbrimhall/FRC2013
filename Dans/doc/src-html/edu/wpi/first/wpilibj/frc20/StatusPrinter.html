005package edu.wpi.first.wpilibj.frc20;
006
007import edu.wpi.first.wpilibj.DriverStationLCD;
008
009/**
010 *
011 * @author freshplum
012 */
013public class StatusPrinter {
014
015    Tray tray;
016
017    StatusPrinter(Tray t) {
018        tray = t;
019    }
020
021    public void printStatuses() {
022        if (tray.isFlywheelLowBattery()) {
023            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser1, 1, "WARNING: Low Battery!");
024        }
025
026        DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser2, 1, "Flywheel: " + tray.flywheelEncoderRate());
027
028        if (tray.isBeltOn()) {
029            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser3, 1, "Belt: On     ");
030        } else if (tray.isBeltReverse()) {
031            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser3, 1, "Belt: Reverse");
032        } else if (tray.isBeltOff()) {
033            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser3, 1, "Belt: Off    ");
034        } else {
035            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser3, 1, "Belt: ?      ");
036        }
037        
038        if(tray.isCollectorOn()) {
039            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser4, 1, "Collector: On     ");
040        }else if(tray.isCollectorReverse()){
041            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser4, 1, "Collector: Reverse");
042        }else{
043            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser4, 1, "Collector: Off    ");
044        }
045        
046        if(tray.isIndexerOut()) {
047            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser5, 1, "Indexer: OUT");
048        }else if(tray.isIndexerIn()) {
049            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser5, 1, "Indexer: In ");
050        }else{
051            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser5, 1, "Indexer: ?  ");
052        }
053        
054        if(tray.isLatchOut()) {
055            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser6, 1, "Stopper: OUT");
056        }else if(tray.isLatchIn()) {
057            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser6, 1, "Stopper: In ");
058        }else {
059            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser6, 1, "Stopper:   ?");
060        }
061
062
063        
064        DriverStationLCD.getInstance().updateLCD();
065    }
066}
